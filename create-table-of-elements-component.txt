 Implementation Details

 Grid Layout Strategy

 - CSS Grid: grid-template-columns: repeat(18, 1fr)
 - Explicit positioning: Use [data-period] and [data-group] attributes with CSS selectors
 - Lanthanides: grid-row: 9; grid-column: 3-17 (atomic numbers 57-71)
 - Actinides: grid-row: 10; grid-column: 3-17 (atomic numbers 89-103)
 - Gaps: Use spacer indicators with asterisk (*) in period 6 and 7 at column 3

 Color Mapping Algorithm

 1. Categorical: Direct lookup from predefined color palette
 2. Continuous:
   - Calculate min/max across all elements for the metric
   - Normalize value to 0-1 range: (value - min) / (max - min)
   - Interpolate through 5-color gradient (viridis-inspired)
 3. Text contrast: Calculate relative luminance (WCAG formula), return white if luminance < 0.5, black otherwise
 4. Performance: Cache colors with key ${atomicNumber}-${metric}, clear cache on metric change

 Responsive Strategy

 - Desktop (>1024px): Full 18-column grid, side drawer for details
 - Tablet (768-1024px): Full 18-column grid with smaller tiles, overlay for details
 - Mobile (<768px): Stacked periods (horizontal scroll per period), bottom sheet for details
 - Touch targets: Minimum 44x44px on mobile

 State Management

   - selectedElement = signal<Element | null>(null)
   - searchQuery = signal<string>('')
   - selectedColorMetric = signal<ColorMetric>('category')

   - legendItems = computed(() => /* legend data for current metric */)

      '#440154', '#3B528B', '#21908C', '#5DC863', '#FDE724'
    ];

    static getCategoricalColor(element: any, metric: string): string {
      switch (metric) {
        case 'category': return this.CATEGORY_COLORS[element.category] || '#6C757D';
        case 'standardState': return this.STATE_COLORS[element.standardState] || '#6C757D';
        case 'block': return this.BLOCK_COLORS[element.block] || '#6C757D';
        default: return '#6C757D';
      }
    }

    static getContinuousColor(value: number | null, min: number, max: number): string {
      if (value === null) return '#2C2C2C'; // Dark gray for missing
      const normalized = (value - min) / (max - min);
      return this.interpolateGradient(Math.max(0, Math.min(1, normalized)));
    }

    private static interpolateGradient(t: number): string {
      const segments = this.GRADIENT_COLORS.length - 1;
      const scaledT = t * segments;
      const index = Math.floor(scaledT);
      const localT = scaledT - index;

      if (index >= segments) return this.GRADIENT_COLORS[segments];

      const color1 = this.hexToRgb(this.GRADIENT_COLORS[index]);
      const color2 = this.hexToRgb(this.GRADIENT_COLORS[index + 1]);

      const r = Math.round(color1.r + (color2.r - color1.r) * localT);
      const g = Math.round(color1.g + (color2.g - color1.g) * localT);
      const b = Math.round(color1.b + (color2.b - color1.b) * localT);

      return this.rgbToHex(r, g, b);
    }

    static calculateTextContrast(bgColorHex: string): string {
      const rgb = this.hexToRgb(bgColorHex);
      const luminance = this.getRelativeLuminance(rgb.r, rgb.g, rgb.b);
      return luminance > 0.5 ? '#000000' : '#FFFFFF';
    }

    private static getRelativeLuminance(r: number, g: number, b: number): number {
      const [rs, gs, bs] = [r, g, b].map(c => {
        const normalized = c / 255;
        return normalized <= 0.03928
          ? normalized / 12.92
          : Math.pow((normalized + 0.055) / 1.055, 2.4);
      });
      return 0.2126 * rs + 0.7152 * gs + 0.0722 * bs;
    }

    private static hexToRgb(hex: string): { r: number; g: number; b: number } {
      const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
      return result ? {
        r: parseInt(result[1], 16),
        g: parseInt(result[2], 16),
        b: parseInt(result[3], 16)
      } : { r: 0, g: 0, b: 0 };
    }

    private static rgbToHex(r: number, g: number, b: number): string {
      return '#' + [r, g, b].map(x => x.toString(16).padStart(2, '0')).join('');
    }
  }

  export class ColorMapper {
    // Retro-futuristic categorical palettes
    private static readonly CATEGORY_COLORS: Record<string, string> = {
      'Alkali Metal': '#FF006E',
      'Alkaline Earth Metal': '#FB5607',
      'Transition Metal': '#8338EC',
      'Post-transition Metal': '#3A86FF',
      'Metalloid': '#06FFA5',
      'Nonmetal': '#FFBE0B',
      'Halogen': '#FF006E',
      'Noble Gas': '#00F5FF',
      'Lanthanide': '#FF69B4',
      'Actinide': '#FFD700'
    };

    private static readonly STATE_COLORS: Record<string, string> = {
      'Solid': '#8338EC',
      'Liquid': '#3A86FF',
      'Gas': '#06FFA5',
      'Unknown': '#6C757D'
    };

    private static readonly BLOCK_COLORS: Record<string, string> = {
      's': '#FF006E',
      'p': '#FFBE0B',
      'd': '#3A86FF',
      'f': '#8338EC'
    };

    // Viridis-inspired gradient for scientific accuracy
    private static readonly GRADIENT_COLORS = [
      '#440154', '#3B528B', '#21908C', '#5DC863', '#FDE724'
    ];

    static getCategoricalColor(element: any, metric: string): string {
      switch (metric) {
        case 'category': return this.CATEGORY_COLORS[element.category] || '#6C757D';
        case 'standardState': return this.STATE_COLORS[element.standardState] || '#6C757D';
        case 'block': return this.BLOCK_COLORS[element.block] || '#6C757D';
        default: return '#6C757D';
      }
    }

    static getContinuousColor(value: number | null, min: number, max: number): string {
      if (value === null) return '#2C2C2C'; // Dark gray for missing
      const normalized = (value - min) / (max - min);
      return this.interpolateGradient(Math.max(0, Math.min(1, normalized)));
    }

    private static interpolateGradient(t: number): string {
      const segments = this.GRADIENT_COLORS.length - 1;
      const scaledT = t * segments;
      const index = Math.floor(scaledT);
      const localT = scaledT - index;

      if (index >= segments) return this.GRADIENT_COLORS[segments];

      const color1 = this.hexToRgb(this.GRADIENT_COLORS[index]);
      const color2 = this.hexToRgb(this.GRADIENT_COLORS[index + 1]);

      const r = Math.round(color1.r + (color2.r - color1.r) * localT);
      const g = Math.round(color1.g + (color2.g - color1.g) * localT);
      const b = Math.round(color1.b + (color2.b - color1.b) * localT);

      return this.rgbToHex(r, g, b);
    }

    static calculateTextContrast(bgColorHex: string): string {
      const rgb = this.hexToRgb(bgColorHex);
      const luminance = this.getRelativeLuminance(rgb.r, rgb.g, rgb.b);
      return luminance > 0.5 ? '#000000' : '#FFFFFF';
    }

    private static getRelativeLuminance(r: number, g: number, b: number): number {
      const [rs, gs, bs] = [r, g, b].map(c => {
        const normalized = c / 255;
        return normalized <= 0.03928
          ? normalized / 12.92
          : Math.pow((normalized + 0.055) / 1.055, 2.4);
      });
      return 0.2126 * rs + 0.7152 * gs + 0.0722 * bs;
    }

    private static hexToRgb(hex: string): { r: number; g: number; b: number } {
      const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
      return result ? {
        r: parseInt(result[1], 16),
        g: parseInt(result[2], 16),
        b: parseInt(result[3], 16)
      } : { r: 0, g: 0, b: 0 };
    }

    private static rgbToHex(r: number, g: number, b: number): string {
      return '#' + [r, g, b].map(x => x.toString(16).padStart(2, '0')).join('');
    }
  }

 Implementation Details

 Grid Layout Strategy

 - CSS Grid: grid-template-columns: repeat(18, 1fr)
 - Explicit positioning: Use [data-period] and [data-group] attributes with CSS selectors
 - Lanthanides: grid-row: 9; grid-column: 3-17 (atomic numbers 57-71)
 - Actinides: grid-row: 10; grid-column: 3-17 (atomic numbers 89-103)
 - Gaps: Use spacer indicators with asterisk (*) in period 6 and 7 at column 3

 Color Mapping Algorithm

 1. Categorical: Direct lookup from predefined color palette
 2. Continuous:
   - Calculate min/max across all elements for the metric
   - Normalize value to 0-1 range: (value - min) / (max - min)
   - Interpolate through 5-color gradient (viridis-inspired)
 3. Text contrast: Calculate relative luminance (WCAG formula), return white if luminance < 0.5, black otherwise
 4. Performance: Cache colors with key ${atomicNumber}-${metric}, clear cache on metric change

 Responsive Strategy

 - Desktop (>1024px): Full 18-column grid, side drawer for details
 - Tablet (768-1024px): Full 18-column grid with smaller tiles, overlay for details
 - Mobile (<768px): Stacked periods (horizontal scroll per period), bottom sheet for details

  export class ColorMapper {
    // Retro-futuristic categorical palettes
    private static readonly CATEGORY_COLORS: Record<string, string> = {
      'Alkali Metal': '#FF006E',
      'Alkaline Earth Metal': '#FB5607',
      'Transition Metal': '#8338EC',
      'Post-transition Metal': '#3A86FF',
      'Metalloid': '#06FFA5',
      'Nonmetal': '#FFBE0B',
      'Halogen': '#FF006E',
      'Noble Gas': '#00F5FF',
      'Lanthanide': '#FF69B4',
      'Actinide': '#FFD700'
    };

    private static readonly STATE_COLORS: Record<string, string> = {
      'Solid': '#8338EC',
      'Liquid': '#3A86FF',
      'Gas': '#06FFA5',
      'Unknown': '#6C757D'
    };

    private static readonly BLOCK_COLORS: Record<string, string> = {
      's': '#FF006E',
      'p': '#FFBE0B',
      'd': '#3A86FF',
      'f': '#8338EC'
    };

    // Viridis-inspired gradient for scientific accuracy
    private static readonly GRADIENT_COLORS = [
      '#440154', '#3B528B', '#21908C', '#5DC863', '#FDE724'
    ];

    static getCategoricalColor(element: any, metric: string): string {
      switch (metric) {
        case 'category': return this.CATEGORY_COLORS[element.category] || '#6C757D';
        case 'standardState': return this.STATE_COLORS[element.standardState] || '#6C757D';
        case 'block': return this.BLOCK_COLORS[element.block] || '#6C757D';
        default: return '#6C757D';
      }
    }

    static getContinuousColor(value: number | null, min: number, max: number): string {
      if (value === null) return '#2C2C2C'; // Dark gray for missing
      const normalized = (value - min) / (max - min);
      return this.interpolateGradient(Math.max(0, Math.min(1, normalized)));
    }

    private static interpolateGradient(t: number): string {
      const segments = this.GRADIENT_COLORS.length - 1;
      const scaledT = t * segments;
      const index = Math.floor(scaledT);
      const localT = scaledT - index;

      if (index >= segments) return this.GRADIENT_COLORS[segments];

      const color1 = this.hexToRgb(this.GRADIENT_COLORS[index]);
      const color2 = this.hexToRgb(this.GRADIENT_COLORS[index + 1]);

      const r = Math.round(color1.r + (color2.r - color1.r) * localT);
      const g = Math.round(color1.g + (color2.g - color1.g) * localT);
      const b = Math.round(color1.b + (color2.b - color1.b) * localT);

      return this.rgbToHex(r, g, b);
    }

    static calculateTextContrast(bgColorHex: string): string {
      const rgb = this.hexToRgb(bgColorHex);
      const luminance = this.getRelativeLuminance(rgb.r, rgb.g, rgb.b);
      return luminance > 0.5 ? '#000000' : '#FFFFFF';
    }

    private static getRelativeLuminance(r: number, g: number, b: number): number {
      const [rs, gs, bs] = [r, g, b].map(c => {
        const normalized = c / 255;
        return normalized <= 0.03928
          ? normalized / 12.92
          : Math.pow((normalized + 0.055) / 1.055, 2.4);
      });
      return 0.2126 * rs + 0.7152 * gs + 0.0722 * bs;
    }

    private static hexToRgb(hex: string): { r: number; g: number; b: number } {
      const result = /^#?([a-f\d]{2})([a-f\d]{2})([a-f\d]{2})$/i.exec(hex);
      return result ? {
        r: parseInt(result[1], 16),
        g: parseInt(result[2], 16),
        b: parseInt(result[3], 16)
      } : { r: 0, g: 0, b: 0 };
    }

    private static rgbToHex(r: number, g: number, b: number): string {
      return '#' + [r, g, b].map(x => x.toString(16).padStart(2, '0')).join('');
    }
  }

 Color Mapping Algorithm

 1. Categorical: Direct lookup from predefined color palette
 2. Continuous:
   - Calculate min/max across all elements for the metric
   - Normalize value to 0-1 range: (value - min) / (max - min)
   - Interpolate through 5-color gradient (viridis-inspired)

 4. Performance: Cache colors with key ${atomicNumber}-${metric}, clear cache on metric change
